176 void Nmer::insertar_cadena(const string & cadena){
177  int hijo;
178  pair<char,int> etiqueta;
179  //posicionamos un nodo, n_act, en la raiz del arbol.
180  ktree<pair<char,int>,4>::node n_act=el_Nmer.root();
181  //Este nodo n lo utilizamos para descender por el arbol
182  // cadena[i] nos indica el índice del nodo en el nivel i+1;
183  // A -> hijo 0; G -> hijo 1; C-> hijo 2; T-> hijo 3
184  //Para cada uno de los caracteres, i, de la cadena
185  for (unsigned int i = 0; i<cadena.size(); i++){
186  //cadena[i] nos indica el índice del nodo en el nivel i+1;
187  // si cadena[i] es : A -> hijo 0; G -> hijo 1; C-> hijo 2; T-> hijo 3
188  switch (cadena[i]) {
189  case 'A': hijo=0; break;
190  case 'G': hijo=1; break;
191  case 'C': hijo=2; break;
192  case 'T': hijo=3; break;
193  }
194  //si n_act tiene hijo en la posición dada por cadena[i] // Ya esta insertado el nodo
195  if (!n_act.k_child(hijo).null()){
196  //incrementamos el contador en 1;
197  (*n_act.k_child(hijo)).second ++;
198  }else{
199  //en caso contrario // No esta insertado el nodo
200  //insertamos el nuevo nodo en dicha posición con etiqueda cadena[i] y su contador a 1;
201  etiqueta.first=cadena[i];
202  etiqueta.second=1;
203  el_Nmer.insert_k_child(n_act,hijo,etiqueta);
204  }
205  //descendemos en el árbol, haciendo que n_act sea el nodo que representa cadena[i], esto es bajamos al
206  //hijo correspondiente;
207  n_act = n_act.k_child(hijo);
208  }
209 }
